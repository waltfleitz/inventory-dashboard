import requests
from bs4 import BeautifulSoup
import socket
import tldextract
import re
import time

HEADERS = {
    "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64)"
}

# -----------------------------------
# DOMAIN VALIDATION
# -----------------------------------
def domain_resolves(url):
    try:
        host = url.replace("https://", "").replace("http://", "").split("/")[0]
        socket.gethostbyname(host)
        return True
    except:
        return False


# -----------------------------------
# CLEAN DOMAIN
# -----------------------------------
def clean_domain(url):
    ext = tldextract.extract(url)
    if ext.domain and ext.suffix:
        return f"https://www.{ext.domain}.{ext.suffix}"
    return None


# -----------------------------------
# GOOGLE SEARCH SCRAPER
# -----------------------------------
def google_search(query):
    try:
        search_url = f"https://www.google.com/search?q={query.replace(' ', '+')}"
        res = requests.get(search_url, headers=HEADERS, timeout=10)

        soup = BeautifulSoup(res.text, "html.parser")
        links = []

        for a in soup.select("a"):
            href = a.get("href", "")
            if "/url?q=" in href:
                link = href.split("/url?q=")[1].split("&")[0]

                # filter junk domains
                if any(x in link for x in [
                    "google", "facebook", "instagram",
                    "twitter", "youtube", "cars.com",
                    "autotrader", "dealerinspire"
                ]):
                    continue

                links.append(link)

        return links

    except Exception as e:
        print(f"[SEARCH ERROR] {e}")
        return []


# -----------------------------------
# SCORE DOMAIN MATCH
# -----------------------------------
def score_domain(link, dealer_name):
    score = 0

    dealer_words = re.sub(r'[^a-zA-Z0-9 ]', '', dealer_name.lower()).split()
    domain = link.lower()

    for word in dealer_words:
        if word in domain:
            score += 1

    return score


# -----------------------------------
# FIND BEST DOMAIN
# -----------------------------------
def find_best_domain(dealer_name, city=None, state=None):
    query = dealer_name
    if city:
        query += f" {city}"
    if state:
        query += f" {state}"

    results = google_search(query)

    if not results:
        return None

    scored = []

    for link in results:
        base = clean_domain(link)
        if not base:
            continue

        score = score_domain(base, dealer_name)
        scored.append((base, score))

    if not scored:
        return None

    scored.sort(key=lambda x: x[1], reverse=True)

    for candidate, score in scored:
        if domain_resolves(candidate):
            return candidate

    return None


# -----------------------------------
# MAIN AUTOCORRECT FUNCTION
# -----------------------------------
def autocorrect_dealer_url(dealer_name, current_url, city=None, state=None):
    print(f"\n[CHECK] {dealer_name}")

    if current_url and domain_resolves(current_url):
        print("  → OK")
        return current_url, "VALID"

    print("  → BROKEN, searching...")

    corrected = find_best_domain(dealer_name, city, state)

    if corrected:
        print(f"  → FIXED: {corrected}")
        return corrected, "FIXED"

    print("  → FAILED")
    return current_url, "DEAD"
